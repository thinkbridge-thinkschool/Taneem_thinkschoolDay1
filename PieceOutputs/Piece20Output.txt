//QuotesIntegrationFactory.cs

public sealed class QuotesIntegrationFactory : WebApplicationFactory<Program>
{
    private readonly string _dbName = Guid.NewGuid().ToString();
    public const string JwtKey =
        "0123456789abcdef0123456789abcdef0123456789abcdef0123456789abcdef";
    public FakeClock Clock { get; } = new();

    protected override void ConfigureWebHost(IWebHostBuilder builder)
    {
        builder.UseEnvironment("Testing");
        builder.ConfigureAppConfiguration((_, config) =>
        {
            config.AddInMemoryCollection(new Dictionary<string, string?>
            {
                ["Jwt:Key"]              = JwtKey,
                ["Jwt:ExpiresInMinutes"] = "60",
                ["Entra:TenantId"]       = "test-tenant",
                ["Entra:ClientId"]       = "test-client",
                ["Entra:Audience"]       = "api://test-client",
            });
        });
        builder.ConfigureServices(services =>
        {
            // Swap SQLite → in-memory EF
            var efDescriptors = services
                .Where(d => d.ServiceType.FullName != null &&
                            (d.ServiceType.FullName.Contains("EntityFrameworkCore") ||
                             d.ServiceType.FullName.Contains("DbContext") ||
                             d.ServiceType == typeof(AppDbContext)))
                .ToList();
            foreach (var d in efDescriptors) services.Remove(d);
            services.AddDbContext<AppDbContext>(opts => opts.UseInMemoryDatabase(_dbName));

            // Swap SystemClock → FakeClock
            var clockDesc = services.SingleOrDefault(d => d.ServiceType == typeof(IClock));
            if (clockDesc != null) services.Remove(clockDesc);
            services.AddSingleton<IClock>(Clock);

            // Override Bearer JWT validation
            services.PostConfigure<JwtBearerOptions>("Bearer", opts =>
            {
                opts.TokenValidationParameters = new TokenValidationParameters
                {
                    ValidateIssuer = false, ValidateAudience = false,
                    ValidateLifetime = true, ValidateIssuerSigningKey = true,
                    IssuerSigningKey = new SymmetricSecurityKey(
                        Encoding.UTF8.GetBytes(JwtKey)),
                    ClockSkew = TimeSpan.Zero,
                };
            });
        });
    }
}

[Fact]
public async Task Login_ValidCredentials_Returns200WithTokens()
{
    _factory.SeedUser("alice@test.com", "Password123!");
    var client = _factory.CreateClient();

    var response = await client.PostAsJsonAsync("/api/auth/login", new
    {
        email    = "alice@test.com",
        password = "Password123!",
    });

    response.StatusCode.Should().Be(HttpStatusCode.OK);
    var body = await response.Content.ReadFromJsonAsync<LoginResponse>(JsonOpts);
    body!.AccessToken.Should().NotBeNullOrEmpty();
    body.RefreshToken.Should().NotBeNullOrEmpty();
    body.ExpiresIn.Should().BePositive();
}

-Error path(Login_WrongPassword_Returns401 (AuthEndpointTests.cs:42))

[Fact]
public async Task Login_WrongPassword_Returns401()
{
    _factory.SeedUser("bob@test.com", "CorrectPassword!");
    var client = _factory.CreateClient();

    var response = await client.PostAsJsonAsync("/api/auth/login", new
    {
        email    = "bob@test.com",
        password = "WrongPassword!",
    });

    response.StatusCode.Should().Be(HttpStatusCode.Unauthorized);
}
