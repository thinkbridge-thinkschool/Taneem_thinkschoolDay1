[Fact]
public void Create_ValidInputs_ReturnsQuoteWithTrimmedValues()
{
    // Arrange
    var author = "  Mark Twain  ";
    var text   = "  The secret of getting ahead is getting started.  ";

    // Act
    var quote = Quote.Create(author, text, createdByUserId: 7);

    // Assert
    quote.Author.Should().Be("Mark Twain");
    quote.Text.Should().Be("The secret of getting ahead is getting started.");
    quote.CreatedByUserId.Should().Be(7);
    quote.IsDeleted.Should().BeFalse();
}

}

Pattern 3:
[Fact]
public void AddItem_WithFakeClock_SetsAddedAtFromClock()
{
    // Arrange
    var fixedTime = new DateTimeOffset(2025, 6, 15, 12, 0, 0, TimeSpan.Zero);

    var clock = Substitute.For<IClock>();
    clock.UtcNow.Returns(fixedTime);   // NSubstitute: always return this value

    var collection = Collection.Create("Test", ownerId: 1);

    // Act
    collection.AddItem(quoteId: 10, clock);

    // Assert
    collection.Items[0].AddedAt.Should().Be(fixedTime);
}
