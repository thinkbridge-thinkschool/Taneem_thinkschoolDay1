1) IClock Interface 
// Services/IClock.cs
namespace QuotesApi.Services;

public interface IClock
{
    DateTimeOffset UtcNow { get; }
}


2) Production Implementation
// Services/IClock.cs (same file)
public sealed class SystemClock : IClock
{
    public DateTimeOffset UtcNow => DateTimeOffset.UtcNow;
}

3)DI Registration 
// Extensions/ServiceExtensions.cs
services.AddSingleton<IClock, SystemClock>();

4)One test using a fake clock
// tests/ClockTest.cs
public sealed class FakeClock : IClock
{
    public DateTimeOffset UtcNow { get; set; }
        = new DateTimeOffset(2026, 1, 1, 0, 0, 0, TimeSpan.Zero);
}

public sealed class ClockTests
{
    [Fact]
    public void AddItem_sets_AddedAt_to_clock_time()
    {
        // Fix the clock to a known time — it will never change
        var clock = new FakeClock
        {
            UtcNow = new DateTimeOffset(2026, 1, 1, 0, 0, 0, TimeSpan.Zero)
        };

        var collection = Collection.Create("My Collection", ownerId: 1);

        collection.AddItem(quoteId: 5, clock);

        // Exact match — no flakiness possible
        Assert.Equal(
            new DateTimeOffset(2026, 1, 1, 0, 0, 0, TimeSpan.Zero),
            collection.Items.Single().AddedAt);
    }
}
